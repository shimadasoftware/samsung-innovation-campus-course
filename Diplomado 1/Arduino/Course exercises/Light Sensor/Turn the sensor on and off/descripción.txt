Materiales:
  2 Resistencias -> 330 Ohm (naranja, naranja, cafe)
  2 Resistencias -> 300 Ohm (cafe, negra, naranja)
  2 Leds -> larga (ánodo +), corta (cátodo -)
  1 pulsador
  1 fotoresistencia
  Cable conectado de 5V a positivo
  Cable conectado de GND a negativo
  Cable conectado de 3 a resistencia
  Cable conectado de 13 a resistencia
  Cable conectado de 7 a pulsador

Nota: para la fotorresistencia trabajar con un número con rayita: -3, -5, -6, etc...

Código:
  int pulsador;

  int luminosidad;
  int brillo;

  void setup() {
    //El A0 no se define como entrada porque siempre es entrada
    Serial.begin(9600); //se configura una comunicacion serial
    Serial.begin(9600); //se configura una comunicacion serial
    pinMode(3, OUTPUT);
    pinMode(13, OUTPUT);
  }

  void loop() {
    luminosidad = analogRead(A0); //en luminosidad se almacena el valor analogico del sensor
    //puede ser un valor entre 0 y 1024
    brillo = map(luminosidad, 54, 1004, 255, 0);

    Serial.println(luminosidad);
    Serial.println(brillo);

    if (luminosidad < 900) {
      digitalWrite(13, HIGH);
    }
    else if (luminosidad > 440) {
      digitalWrite(13, LOW);
    }
    analogWrite(3, brillo);
  }
