print 'Welcome to exercise #01!'

print 'Please, enter a number:'
number = input_int()

count = 1
for i in range(number):
    for j in range(number - i):
        robot = UsedRobot(
            avenues = (j + count),
            streets = i + 1,
            beepers = number - i,
            colour = 'purple',
            name = 'robot1'
        )
        robot.put_beeper()
    count += 1
robot.turn_off()
